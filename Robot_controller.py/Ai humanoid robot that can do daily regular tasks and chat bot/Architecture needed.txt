STAGE 4: VOICE CONTROL

We add speech → commands.

Python Voice Node

</> Python

import speech_recognition as sr
import rclpy
from std_msgs.msg import String

def listen_and_publish():
    rclpy.init()
    node = rclpy.create_node('voice_node')
    pub = node.create_publisher(String, 'robot_commands', 10)

    r = sr.Recognizer()
    mic = sr.Microphone()

    with mic as source:
        print("Listening...")
        audio = r.listen(source)

    try:
        text = r.recognize_google(audio)
        print("You said:", text)

        msg = String()
        msg.data = text
        pub.publish(msg)

    except:
        print("Speech not recognized")

import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)

MOTOR_PIN = 18
GPIO.setup(MOTOR_PIN, GPIO.OUT)

GPIO.output(MOTOR_PIN, True)
time.sleep(2)
GPIO.output(MOTOR_PIN, False)

GPIO.cleanup()
